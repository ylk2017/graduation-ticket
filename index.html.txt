function makeQRDataUrl(text, size = 400) {
  return new Promise(resolve => {
    const tmp = document.createElement('div');
    tmp.style.display = 'none';
    document.body.appendChild(tmp);
    new QRCode(tmp, {
      text, width: size, height: size,
      colorDark: '#000000', colorLight: '#ffffff',
      correctLevel: QRCode.CorrectLevel.H
    });
    setTimeout(() => {
      const canvas = tmp.querySelector('canvas');
      const url = canvas.toDataURL('image/png');
      tmp.remove();
      resolve(url);
    }, 80);
  });
}

async function generateStudentPDF(studentId) {
  const qr1 = await makeQRDataUrl(`${studentId}-T1`);
  const qr2 = await makeQRDataUrl(`${studentId}-T2`);
  const ticket1Img = await drawTicketCanvas(studentId, 'T1', qr1);
  const ticket2Img = await drawTicketCanvas(studentId, 'T2', qr2);

  const { jsPDF } = window.jspdf;
  const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' });

  const ticketW = 190;
  const ticketH = ticketW * (700 / 1200);
  const x = (210 - ticketW) / 2;

  pdf.addImage(ticket1Img, 'PNG', x, 15, ticketW, ticketH);

  const cutY = 15 + ticketH + 12;
  pdf.setDrawColor(180, 180, 180);
  pdf.setLineDashPattern([3, 2], 0);
  pdf.line(10, cutY, 200, cutY);
  pdf.setLineDashPattern([], 0);
  pdf.setFontSize(9);
  pdf.setTextColor(150, 150, 150);
  pdf.text('- - - Cut here - - -', 105, cutY - 2, { align: 'center' });

  pdf.addImage(ticket2Img, 'PNG', x, cutY + 10, ticketW, ticketH);

  return pdf;
}

async function downloadSinglePDF(id) {
  const pdf = await generateStudentPDF(id);
  pdf.save(`${id}.pdf`);
}

async function downloadAllPDFs() {

}

async function downloadFilteredPDFs() {
  const filter = document.getElementById('classFilter').value;
  const list = filter === 'all' ? STUDENTS : STUDENTS.filter(s => s.startsWith(filter));
  const name = filter === 'all' ? 'All_Tickets.zip' : `${filter}_Tickets.zip`;
  await downloadBatch(list, name);
}

async function downloadBatch(list, zipName) {
  const bar = document.getElementById('progressBar');
  const fill = document.getElementById('progressFill');
  bar.style.display = 'block';
  const zip = new JSZip();
  for (let i = 0; i < list.length; i++) {
    const id = list[i];
    const pdf = await generateStudentPDF(id);
    const blob = pdf.output('blob');
    zip.file(`${id}.pdf`, blob);
    const pct = Math.round(((i + 1) / list.length) * 100);
    fill.style.width = pct + '%';
    fill.textContent = `${pct}% (${i + 1}/${list.length})`;
  }
  const zipBlob = await zip.generateAsync({ type: 'blob' });
  const url = URL.createObjectURL(zipBlob);
  const a = document.createElement('a');
  a.href = url; a.download = zipName; a.click();
  URL.revokeObjectURL(url);
  setTimeout(() => { bar.style.display = 'none'; fill.style.width = '0%'; }, 1500);
}

let html5QrCode = null;
async function startScan() {
  document.getElementById('scanResult').style.display = 'none';
  try {
    html5QrCode = new Html5Qrcode('reader');
    await html5QrCode.start(
      { facingMode: 'environment' },
      { fps: 10, qrbox: { width: 250, height: 250 } },
      (decoded) => { verifyTicket(decoded, '鏡頭掃描', 'scanResult'); },
      () => {}
    );
    document.getElementById('startScanBtn').style.display = 'none';
    document.getElementById('stopScanBtn').style.display = 'inline-block';
  } catch (e) {
    showResult('scanResult', 'error', '❌ 無法啟動鏡頭：' + e.message + '<br>請檢查：1) 使用 HTTPS 2) 允許鏡頭權限');
  }
}

async function stopScan() {
  if (html5QrCode) {
    try { await html5QrCode.stop(); html5QrCode.clear(); } catch(e){}
    html5QrCode = null;
  }
  document.getElementById('startScanBtn').style.display = 'inline-block';
  document.getElementById('stopScanBtn').style.display = 'none';
}

function scanImage(evt) {
  const file = evt.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    const img = new Image();
    img.onload = () => {
      const canvas = document.createElement('canvas');
      canvas.width = img.width; canvas.height = img.height;
      const ctx = canvas.getContext('2d');
      ctx.drawImage(img, 0, 0);
      const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height);
      const code = jsQR(imgData.data, imgData.width, imgData.height);
      if (code) {
        verifyTicket(code.data, '圖片上傳', 'imageResult');
      } else {
        showResult('imageResult', 'error', '❌ 無法識別此圖片中的 QR Code，請使用更清晰的圖片。');
      }
    };
    img.src = e.target.result;
  };
  reader.readAsDataURL(file);
}

function verifyManual() {
  const val = document.getElementById('manualInput').value.trim().toUpperCase();
  if (!val) { showResult('manualResult', 'error', '⚠️ 請輸入票券編號'); return; }
  verifyTicket(val, '手動輸入', 'manualResult');
  document.getElementById('manualInput').value = '';
}

function verifyTicket(code, method, resultElId) {
  code = code.trim().toUpperCase();
  if (!VALID_TICKETS.has(code)) {
    showResult(resultElId, 'error', `❌ 無效票券：${code}<br>此票券不存在`);
    return;
  }
  const existing = records.find(r => r.ticket === code);
  if (existing) {
    showResult(resultElId, 'warning', `⚠️ ${code} 已於 ${existing.time} 使用！<br>（此票券已重複）`);
    return;
  }
  const time = new Date().toLocaleString('zh-HK', { hour12: false });
  records.push({ ticket: code, time, method });
  const [sid, tno] = code.split('-');
  showResult(resultElId, 'success', `✅ 歡迎入場！<br><span style="font-size:1.4em;">${sid}</span> · 票券 ${tno}<br>${time}`);
}

function showResult(elId, type, html) {
  const el = document.getElementById(elId);
  el.className = 'scan-result ' + type;
  el.innerHTML = html;
  el.style.display = 'block';
}

function renderRecords() {
  const tbody = document.getElementById('recordsTable');
  tbody.innerHTML = '';
  if (records.length === 0) {
    tbody.innerHTML = '<tr><td colspan="4" style="text-align:center; padding:20px; color:#999;">尚無入場記錄</td></tr>';
  } else {
    records.slice().reverse().forEach((r, i) => {
      const tr = document.createElement('tr');
      tr.innerHTML = `<td>${records.length - i}</td><td><b>${r.ticket}</b></td><td>${r.time}</td><td>${r.method}</td>`;
      tbody.appendChild(tr);
    });
  }
  document.getElementById('statTotal').textContent = records.length;
  ['6A','6B','6C','6D','6S'].forEach(c => {
    document.getElementById('stat' + c).textContent = records.filter(r => r.ticket.startsWith(c)).length;
  });
}

function exportCSV() {
  if (records.length === 0) { alert('尚無記錄可匯出'); return; }
  let csv = '\uFEFF票券編號,入場時間,方式\n';
  records.forEach(r => { csv += `${r.ticket},${r.time},${r.method}\n`; });
  const blob = new Blob([csv], { type: 'text/csv' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = `入場記錄_${new Date().toISOString().slice(0,10)}.csv`;
  a.click();
  URL.revokeObjectURL(url);
}

function clearRecords() {
  if (confirm('確定要清除所有入場記錄嗎？此操作無法復原。')) {
    records.length = 0;
    renderRecords();
  }
}
