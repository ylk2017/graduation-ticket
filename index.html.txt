function scanImage(evt) {
  const file = evt.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    const img = new Image();
    img.onload = () => {
      const canvas = document.createElement('canvas');
      canvas.width = img.width; canvas.height = img.height;
      const ctx = canvas.getContext('2d');
      ctx.drawImage(img, 0, 0);
      const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height);
      const code = jsQR(imgData.data, imgData.width, imgData.height);
      if (code) {
        verifyTicket(code.data, '圖片上傳', 'imageResult');
      } else {
        showResult('imageResult', 'error', '❌ 無法識別此圖片中的 QR Code');
      }
    };
    img.src = e.target.result;
  };
  reader.readAsDataURL(file);
}

/* ✅ 修改：驗票時寫入 Firebase，用 transaction 確保不會被多部機重複 scan */
function verifyTicket(code, method, resultElId) {
  code = code.trim().toUpperCase();
  if (!VALID_TICKETS.has(code)) {
    showResult(resultElId, 'error', `❌ 無效票券：${code}<br>此票券不存在`);
    return;
  }

  if (!recordsRef) {
    // 離線模式 fallback
    if (records.has(code)) {
      const existing = records.get(code);
      showResult(resultElId, 'warning', `⚠️ ${code} 已於 ${existing.time} 使用！（離線）`);
      return;
    }
    const time = new Date().toLocaleString('zh-HK', { hour12: false });
    records.set(code, { time, method, teacher: myTeacherName || '未命名' });
    const [sid, tno] = code.split('-');
    showResult(resultElId, 'success', `✅ 歡迎入場！（離線）<br><span style="font-size:1.4em;">${sid}</span> · 票券 ${tno}<br>${time}`);
    updateDashboardStats();
    return;
  }

  // 即時檢查（快速反應）
  if (records.has(code)) {
    const existing = records.get(code);
    showResult(resultElId, 'warning',
      `⚠️ ${code} 已於 ${existing.time} 使用！<br>` +
      `驗票員：${existing.teacher || '未命名'}（${existing.method || '-'}）`);
    return;
  }

  // 使用 Firebase transaction 確保同時有多部機掃同一張票時只有一次成功
  const time = new Date().toLocaleString('zh-HK', { hour12: false });
  const info = { time, method, teacher: myTeacherName || '未命名' };

  recordsRef.child(code).transaction(
    current => {
      if (current !== null) return; // 已被其他裝置用了 → abort
      return info;
    },
    (error, committed, snapshot) => {
      if (error) {
        showResult(resultElId, 'error', `❌ 雲端錯誤：${error.message}`);
      } else if (!committed) {
        const existing = snapshot.val();
        showResult(resultElId, 'warning',
          `⚠️ ${code} 已於 ${existing.time} 使用！<br>` +
          `驗票員：${existing.teacher || '未命名'}`);
      } else {
        const [sid, tno] = code.split('-');
        showResult(resultElId, 'success',
          `✅ 歡迎入場！<br><span style="font-size:1.4em;">${sid}</span> · 票券 ${tno}<br>${time}`);
      }
    }
  );
}

          <span class="${t2 ? 'used' : ''}">T2</span>
        </div>
      `;
      grid.appendChild(cell);
    });
  }

  updateDashboardStats();
}

let currentModalSid = null;
function openModal(sid) {
  currentModalSid = sid;
  document.getElementById('modalStudentId').textContent = sid;
  renderModalTickets();
  document.getElementById('modalBackdrop').classList.add('active');
}

function closeModal() {
  document.getElementById('modalBackdrop').classList.remove('active');
  currentModalSid = null;
}

function renderModalTickets() {
  const sid = currentModalSid;
  if (!sid) return;
  const container = document.getElementById('modalTicketsList');
  container.innerHTML = '';
  ['T1', 'T2'].forEach(tn => {
    const code = `${sid}-${tn}`;
    const used = records.has(code);
    const info = used ? records.get(code) : null;
    const row = document.createElement('div');
    row.className = 'ticket-row';
    row.innerHTML = `
      <div>
        <div style="font-weight:bold; font-size:1.1em;">${code}</div>
        <div style="font-size:0.8em; color:#666; margin-top:2px;">
          ${used ? `✓ ${info.time} (${info.method})${info.teacher ? ' · '+info.teacher : ''}` : '尚未入場'}
        </div>
      </div>
      <div>
        <span class="tag ${used ? 'used' : 'unused'}">${used ? '已入場' : '未入場'}</span>
        ${used
          ? `<button class="btn btn-danger btn-sm" onclick="revokeTicket('${code}')">↩️ 收回</button>`
          : `<button class="btn btn-success btn-sm" onclick="markTicket('${code}')">✅ 標記入場</button>`
        }
      </div>
    `;
    container.appendChild(row);
  });
}

/* ✅ 修改：手動標記寫入 Firebase */
function markTicket(code) {
  const time = new Date().toLocaleString('zh-HK', { hour12: false });
  const info = { time, method: '手動標記', teacher: myTeacherName || '未命名' };
  if (recordsRef) {
    recordsRef.child(code).set(info);
  } else {
    records.set(code, info);
    renderModalTickets();
    renderDashboard();
  }
}

/* ✅ 修改：收回記錄同步到 Firebase */
function revokeTicket(code) {
  if (confirm(`確定要收回 ${code} 的入場記錄嗎？\n所有裝置都會同步此變更。`)) {
    if (recordsRef) {
      recordsRef.child(code).remove();
    } else {
      records.delete(code);
      renderModalTickets();
      renderDashboard();
    }
  }
}

function exportCSV() {
  if (records.size === 0) { alert('尚無記錄可匯出'); return; }
  let csv = '\uFEFF票券編號,學生編號,票號,入場時間,方式,驗票員\n';
  const sorted = Array.from(records.entries()).sort((a,b) => a[0].localeCompare(b[0]));
  sorted.forEach(([code, info]) => {
    const [sid, tno] = code.split('-');
    csv += `${code},${sid},${tno},${info.time},${info.method},${info.teacher || ''}\n`;
  });
  const blob = new Blob([csv], { type: 'text/csv' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = `入場記錄_${new Date().toISOString().slice(0,10)}.csv`;
  a.click();
  URL.revokeObjectURL(url);
}

/* ✅ 修改：清除記錄同步到 Firebase */
function clearRecords() {
  if (confirm('⚠️ 確定要清除所有入場記錄嗎？\n此操作會同步清除所有裝置的記錄，無法復原！')) {
    if (recordsRef) {
      recordsRef.remove();
    } else {
      records.clear();
      renderDashboard();
    }
  }
}
